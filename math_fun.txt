#include<iostream>
#include<cmath>
using namespace std;
int main()
{
    int pi=3.14;
    cout<<"cos(60):\n"<<cos(60*pi/180.0)<<endl;
    cout<<"sin(60):\n"<<sin(60*pi/180.0)<<endl;
    cout<<"2^3:\n"<<pow(2,3)<<endl;
    cout<<"Square root of 100:\n"<<sqrt(100)<<endl;
    
    cout<<"log of 5:\n"<<log(5)<<endl;
    return 0;
}
